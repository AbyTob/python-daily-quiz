class ThreadSafeCounter:
    def __init__(self):
        self.value = 0
        self.lock = threading.Lock()

            return self._value

            self._value = new_value

def increment_counter(counter):

        counter.value += 1

threads = [threading.Thread(target=increment_counter, args=(counter,)) for _ in range(10)]
for thread in threads:
    thread.start()
