```python
class AsyncTimer:
    def __init__(self, interval):
        self.interval = interval
        self.tasks = []

        while True:

            for task in self.tasks:
                if not task.done():
                    await task

    def add_task(self, task):
        self.tasks.append(task)

async def my_async_task(timer):

    await asyncio.sleep(2)
    print("Task completed")

async def main():
    timer = AsyncTimer(interval=1)
    timer.add_task(my_async_task(timer))
    await timer.start()

# Run the example
asyncio.run(main())
```
