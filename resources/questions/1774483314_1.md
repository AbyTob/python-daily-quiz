```python
import time

class Timer:
    def __init__(self):
        self.start_time = None
    
    def start(self):
        if not self.start_time:
            self.start_time = time.time()
    
    def stop(self):
        return time.time() - self.start_time

def timer_decorator(func):
    def wrapper(*args, **kwargs):
        t = Timer()
        t.start()
        result = func(*args, **kwargs)
        t.stop()
        print(f"Function {func.__name__} took {t.stop()} seconds to run.")
        return result
    return wrapper

@timer_decorator
def compute_sum(n):
    return sum(range(n))

print(compute_sum(1000000))
```

A) The `Timer` class is a context manager.

C) The `compute_sum` function will run faster when decorated with `@timer_decorator`.  
D) The output of the code includes the total time taken by the `compute_sum` function.
