```python
from abc import ABC, abstractmethod
from contextlib import contextmanager
import time

class TimerMeta(type):
    def __new__(cls, name, bases, dct):
        for attr_name, attr in dct.items():
            if callable(attr) and not attr_name.startswith("__"):
                dct[attr_name] = cls.wrap_timer_method(attr)
        return super().__new__(cls, name, bases, dct)

    @staticmethod
    def wrap_timer_method(method):
        def wrapper(self, *args, **kwargs):
            start_time = time.time()
            result = method(self, *args, **kwargs)
            end_time = time.time()
            print(f"Executing {method.__name__} took {end_time - start_time:.4f}s")
            return result
        return wrapper

class BaseTask(metaclass=TimerMeta):
    @abstractmethod
    def execute(self):
        pass

@contextmanager
def task_manager():
    try:
        yield
    finally:
        print("All tasks have been executed.")

class ConcreteTask(BaseTask):
    def execute(self):
        time.sleep(1)
        print("Executing task")

# Usage
with task_manager():
    task = ConcreteTask()
    task.execute()
```

D) The `TimerMeta` metaclass can be used to create subclasses of `BaseTask` without adding any additional functionality.
