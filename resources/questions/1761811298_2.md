    def __new__(cls, name, bases, dct):
        for attr_name, attr_value in dct.items():

                dct[attr_name] = cls.log_method_call(attr_value)
        return super().__new__(cls, name, bases, dct)

    @staticmethod
    def log_method_call(method):
